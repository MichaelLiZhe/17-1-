public class xuliehuahua implements Serializable{
	private String name;
	private String age;
	
	public xuliehuahua(){
		super();
		
	}
	public xuliehuahua(String name,String age){
		super();
		this.name=name;
		this.age=age;

		this.name = name;
	}
	public String getAge() {
		return age;
	}

		this.age = age;
	}
	

}
